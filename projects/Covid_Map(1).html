# Define the function to update the plot
def update_plot(date,column):

    fig, ax = plt.subplots(figsize=(20, 6))

    # Clear the previous plot
    ax.clear()

    # Display the slider widget and initial plot
    worldmap = gpd.read_file(gpd.datasets.get_path("naturalearth_lowres"))
    worldmap.plot(color="lightgrey", ax=ax)

    # Filter the data based on the selected day
    filtered_df = df[df['Date']== date ]

    # Plot the filtered data
    x = filtered_df['Long']
    y = filtered_df['Lat']
    z = filtered_df[column]

    scatter = plt.scatter(x, y, s=z/200, c=z, alpha=0.6, cmap='autumn')

    plt.colorbar(scatter,shrink=1)

    plt.xlabel('Longitude')
    plt.ylabel('Latitude')
    plt.title("COVID-19 Deaths on {}".format(date.strftime("%Y-%m-%d")))

    plt.xlim([-180, 180])
    plt.ylim([-90, 90])
    plt.show()

# Create date slider
dates = pd.date_range(start=df["Date"].min(), end=df["Date"].max(), freq='D')
date_slider = widgets.SelectionSlider(options=dates, 
                                      value= dates[-1], #to start on the last day
                                      description='Date', 
                                      continuous_update=False)
# Creat a Dropdown menu
Column_Values = ['Recovered','Deaths']

dropdown=widgets.Dropdown(
    options=Column_Values,
    value=Column_Values[0],
    description='Column:',
    disabled=False,
)

# Link slider to update function
widgets.interact(update_plot, date=date_slider,column=dropdown)
